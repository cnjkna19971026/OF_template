import json
import os
import sys

# --- CONFIGURATION ---
QUALITY_MAP = {
    "standard": "(1 1)",
    "medium":   "(2 2)",
    "high":     "(3 3)",
    "ultra":    "(4 4)"
}

def load_config():
    try:
        with open("config.json", "r") as f:
            return json.load(f)
    except FileNotFoundError:
        print("Error: config.json not found.")
        sys.exit(1)

def get_stl_centroid(stl_path):
    """
    Reads an ASCII STL file and returns the center (x, y, z) of its bounding box.
    Does not require numpy.
    """
    min_x, max_x = float('inf'), float('-inf')
    min_y, max_y = float('inf'), float('-inf')
    min_z, max_z = float('inf'), float('-inf')
    
    vertex_count = 0
    
    try:
        with open(stl_path, 'r') as f:
            for line in f:
                if "vertex" in line:
                    parts = line.strip().split()
                    # vertex x y z -> parts[1], parts[2], parts[3]
                    x = float(parts[1])
                    y = float(parts[2])
                    z = float(parts[3])
                    
                    if x < min_x: min_x = x
                    if x > max_x: max_x = x
                    if y < min_y: min_y = y
                    if y > max_y: max_y = y
                    if z < min_z: min_z = z
                    if z > max_z: max_z = z
                    vertex_count += 1
    except Exception as e:
        print(f"Error reading STL {stl_path}: {e}")
        return None

    if vertex_count == 0:
        return None

    # Calculate center
    center_x = (min_x + max_x) / 2
    center_y = (min_y + max_y) / 2
    center_z = (min_z + max_z) / 2
    
    return [center_x, center_y, center_z]

def generate_snappy(data):
    print("--- Generating system/snappyHexMeshDict ---")
    
    components = data["components"]
    
    # --- AUTO-CALCULATE LOCATION IN MESH ---
    # We try to find the 'box' component to calculate the center of the domain
    location_pt = [0.01, 0.01, 0.01] # Default fallback
    found_box = False
    back_ground_file = "air_small"


    for comp in components:
        if back_ground_file in comp["name"].lower() or "case" in comp["name"].lower():
            print(f"Detected main case component: {comp['stl_file']}")
            calculated_center = get_stl_centroid(f"constant/triSurface/{comp['stl_file']}")
            if calculated_center:
                location_pt = calculated_center
                found_box = True
                print(f"Auto-calculated 'locationInMesh' (Air Point): {location_pt}")
            break
    
    if not found_box:
        print(f"Warning: Could not identify a {back_ground_file} component. Using default locationInMesh.")

    location_str = f"({location_pt[0]:.4f} {location_pt[1]:.4f} {location_pt[2]:.4f})"

    # --- WRITING THE DICTIONARY ---
    lines = []
    lines.append("FoamFile { version 2.0; format ascii; class dictionary; object snappyHexMeshDict; }")
    lines.append("// Auto-generated by setup_case.py\n")
    
    lines.append("castellatedMesh true;")
    lines.append("snap true;")
    lines.append("addLayers false;")
    
    lines.append("geometry")
    lines.append("{")
    for comp in components:
        lines.append(f"    {comp['name']}")
        lines.append(f"    {{ type triSurfaceMesh; file \"{comp['stl_file']}\"; }}")
    lines.append("};\n")

    lines.append("castellatedMeshControls")
    lines.append("{")
    lines.append("    maxLocalCells 1000000; maxGlobalCells 2000000; minRefinementCells 10; maxLoadUnbalance 0.10; nCellsBetweenLevels 3;")
    lines.append("    features ();")
    
    lines.append("    refinementSurfaces")
    lines.append("    {")
    for comp in components:
        level = QUALITY_MAP.get(comp["quality"], "(2 2)")
        lines.append(f"        {comp['name']}")
        lines.append("        {")
        lines.append(f"            level {level};")
        lines.append(f"            faceZone {comp['name']}_face;")
        lines.append(f"            cellZone {comp['name']};")
        lines.append(f"            zoneInside true;")
        lines.append("        }")
    lines.append("    }")
    
    lines.append("    resolveFeatureAngle 30;")
    lines.append("    refinementRegions {}")
    lines.append(f"    locationInMesh {location_str};") # <--- THIS IS NOW AUTOMATIC
    lines.append("    allowFreeStandingZoneFaces true;")
    lines.append("}\n")

    lines.append("snapControls { nSmoothPatch 3; tolerance 2.0; nSolveIter 30; nRelaxIter 5; }")
    lines.append("addLayersControls { relativeSizes true; layers {} }")
    lines.append("meshQualityControls { #include \"meshQualityDict\" }")
    lines.append("mergeTolerance 1e-6;")

    os.makedirs("system", exist_ok=True)
    with open("system/snappyHexMeshDict", "w") as f:
        f.write("\n".join(lines))
    print("System dictionary written.")

# (Function generate_heat_sources is same as before, included for completeness)
def generate_heat_sources(data):
    print("--- Generating Heat Sources ---")
    components = data["components"]
    lines = []
    lines.append("FoamFile { version 2.0; format ascii; class dictionary; object fvOptions; }")
    lines.append("options {")

    for comp in components:
        power = comp.get("power_watts", 0)
        if power > 0:
            lines.append(f"    heatSource_{comp['name']}")
            lines.append("    { type scalarSemiImplicitSource; active on; selectionMode cellZone;")
            lines.append(f"      cellZone {comp['name']}; volumeMode absolute;")
            lines.append(f"      sources {{ h {power}; }} }}")

    lines.append("}")
    with open("system/fvOptions", "w") as f:
        f.write("\n".join(lines))


def generate_decompose_par(data):
    print("--- Generating system/decomposeParDict ---")
    
    # Default to 4 cores if not specified
    cores = data.get("case_settings", {}).get("cores", 4)
    
    lines = []
    lines.append("FoamFile { version 2.0; format ascii; class dictionary; location \"system\"; object decomposeParDict; }")
    lines.append("// Auto-generated by setup_case.py\n")
    
    lines.append(f"numberOfSubdomains {cores};")
    
    # 'scotch' is the best method for automation. 
    # It requires zero geometric input from you.
    lines.append("method scotch;")
    
    # Optional: Keep these empty for standard scotch behavior
    lines.append("scotchCoeffs")
    lines.append("{")
    lines.append("    // processorWeights ( ... ); // Add weights if using mixed hardware")
    lines.append("}")
    
    # For CHT, we sometimes need to handle constraints, but usually 
    # default scotch works fine for multi-region.
    lines.append("constraints")
    lines.append("{")
    lines.append("}")

    lines.append("distributed no;")
    lines.append("roots ( );")

    with open("system/decomposeParDict", "w") as f:
        f.write("\n".join(lines))
    print(f"   [OK] Configured for {cores} cores using 'scotch' method.")

# --- DONT FORGET TO ADD THE CALL IN __main__ ---
if __name__ == "__main__":
    data = load_config()
    generate_snappy(data)
    generate_heat_sources(data)
    generate_decompose_par(data) # <--- Add this call
    print("\nSUCCESS: Setup complete.")
